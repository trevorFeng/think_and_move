### 遍历算法
- 先序
    - 访问根节点
    - 访问左节点
    - 访问右节点
    
- 中序遍历
    - 访问左节点
    - 访问根节点
    - 访问右节点
  
- 后序遍历
    - 访问左节点
    - 访问右节点
    - 访问根节点
   
```java
package aaaTree;

import java.util.*;

public class TestTree {

    /**
     * 二叉树的构建
     */
    public void buildTree(TreeNode node, int data) {
        if (node == null) {
            node = new TreeNode(data);
        }else {
            if (data < node.val) {
                if (node.left == null) {
                    node.left = new TreeNode(data);
                }else {
                    buildTree(node.left, data);
                }
            }else {
                if (node.right == null) {
                    node.right = new TreeNode(data);
                }else {
                    buildTree(node.right, data);
                }
            }
        }

    }

    /**
     * 先序遍历
     */
    public void xianXu(TreeNode treeNode ){
        Stack<TreeNode> stack = new Stack<>();
        TreeNode p = treeNode;
        while (!stack.isEmpty() || p != null) {
            if (p != null) {
                System.out.println(p.val);
                stack.push(p);
                p = p.left;
            }else {
                p = stack.pop();
                p = p.right;
            }
        }
    }

    /**
     * 中序遍历
     *
     */
    public void zhongXu(TreeNode node) {
        TreeNode p = node;
        Stack<TreeNode> stack = new Stack<>();
        while (p != null || !stack.isEmpty()) {
            if (p != null) {
                stack.push(p);
                p = p.left;
            }else {
                p = stack.pop();
                System.out.println(p.val);
                p = p.right;
            }
        }
    }

    /**
     * 后续遍历
     */
    public void houXu(TreeNode node) {
        TreeNode p = node;
        TreeNode b = null;
        Stack<TreeNode> stack = new Stack<>();
        while(p != null || !stack.isEmpty()) {
            if (p != null) {
                stack.push(p);
                p = p.left;
            }else {
               p = stack.peek();
                if (p.right != null || p.right != b) {
                    p = p.right;
                    stack.push(p);
                    p = p.left;
                }else {
                     p = stack.pop();
                     System.out.println(p.val);
                     b = p;
                     p = null;
                }
            }
        }
    }

    /**
     * 广度优先bfs
     */
    public void bfs(TreeNode t) {
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(t);
        while (!queue.isEmpty()) {
            TreeNode treeNode = queue.poll();
            System.out.println(treeNode.val);
            if (treeNode.left != null) {
                queue.offer(treeNode.left);
            }
            if (treeNode.right != null) {
                queue.offer(treeNode.right);
            }
        }
    }

    /**
     * 深度优先dfs，回溯法
     */
    public void dfs(TreeNode t, List<List<Integer>> rst, List<Integer> list) {
        if (t == null) {
            return;
        }
        list.add(t.val);
        if (t.right == null && t.left == null) {
            list.add(t.val);
            rst.add(new ArrayList<>(list));
            list.remove(list.size() - 1);
        }
        dfs(t.right, rst, list);
        dfs(t.left, rst, list);
        list.remove(list.size() - 1);
    }


}

class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    public TreeNode(int val) {
        this.val = val;
    }
}
```
