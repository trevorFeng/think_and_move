给一个排序二叉树，求出比target小的最大值,没有返回null
```
public int test(TreeNode node, int num){
    if(node == null){
        return null;
    }
    if(node.val>=num){
        return test(node.left, num);
    }else{
        int result = test(node.right, num);
        return result == null ? node.value : result;
    }
}
```
